187bool I2C_writeBits(uint8_t devAddr, uint8_t regAddr, uint8_t bitStart, uint8_t length, uint8_t data) {
188 // 010 value to write
189 // 76543210 bit numbers
190 // xxx args: bitStart=4, length=3
191 // 00011100 mask byte
192 // 10101111 original value (sample)
193 // 10100011 original & ~mask
194 // 10101011 masked | value
195 uint8_t b = 0;
196 if (I2C_readByte(devAddr, regAddr, &b, 0) != 0) {
197 uint8_t mask = ((1 << length) - 1) << (bitStart - length + 1);
198 data <<= (bitStart - length + 1); // shift data into correct position
199 data &= mask; // zero all non-important bits in data
200 b &= ~(mask); // zero all important bits in existing byte
201 b |= data; // combine data with existing byte
202 return I2C_writeByte(devAddr, regAddr, b);
203 } else {
204 return false;
205 }
206}
