25void MPU6050_ReadRegister(uint8_t reg, uint8_t *data, uint8_t len){
26 uint8_t dev = 0x68;
27 i2c_cmd_handle_t cmd;
28 I2C_SelectRegister(dev, reg);
29
30 cmd = i2c_cmd_link_create();
31 ESP_ERROR_CHECK(i2c_master_start(cmd));
32 ESP_ERROR_CHECK(i2c_master_write_byte(cmd, (dev << 1) | I2C_MASTER_READ, 1));
33
34 if(len>1)
35 ESP_ERROR_CHECK(i2c_master_read(cmd, data, len, I2C_MASTER_ACK));
36
37 ESP_ERROR_CHECK(i2c_master_read_byte(cmd, data+len-1, I2C_MASTER_NACK));
38
39 ESP_ERROR_CHECK(i2c_master_stop(cmd));
40 ESP_ERROR_CHECK(i2c_master_cmd_begin(I2C_NUM, cmd, 1000));
41 i2c_cmd_link_delete(cmd);
42}
