67void NeoPixelSetArray(neopixel_color_t *color_array){
68 rgb_led_t led;
69 uint16_t red, green, blue;
70 for (uint16_t i = 0; i < stripe_length; i++){
71 red = ((color_array[i] & RED_MSK) >> RED_OFFSET) * stripe_bright;
72 green = ((color_array[i] & GREEN_MSK) >> GREEN_OFFSET) * stripe_bright;
73 blue = ((color_array[i] & BLUE_MSK) >> BLUE_OFFSET) * stripe_bright;
74 led.red = red >> BRIGHT_OFFSET;
75 led.green = green >> BRIGHT_OFFSET;
76 led.blue = blue >> BRIGHT_OFFSET;
77 ws2812bSend(led);
78 }
79 ws2812bSendRet();
80}

119neopixel_color_t NeoPixelHSV2Color(uint16_t hue, uint8_t sat, uint8_t val){
120
121 uint8_t r, g, b;
122
123 hue = (hue * 1530L + 32768) / 65536;
124 // Convert hue to R,G,B (nested ifs faster than divide+mod+switch):
125 if (hue < 510) { // Red to Green-1
126 b = 0;
127 if (hue < 255) { // Red to Yellow-1
128 r = 255;
129 g = hue; // g = 0 to 254
130 } else { // Yellow to Green-1
131 r = 510 - hue; // r = 255 to 1
132 g = 255;
133 }
134 } else if (hue < 1020) { // Green to Blue-1
135 r = 0;
136 if (hue < 765) { // Green to Cyan-1
137 g = 255;
138 b = hue - 510; // b = 0 to 254
139 } else { // Cyan to Blue-1
140 g = 1020 - hue; // g = 255 to 1
141 b = 255;
142 }
143 } else if (hue < 1530) { // Blue to Red-1
144 g = 0;
145 if (hue < 1275) { // Blue to Magenta-1
146 r = hue - 1020; // r = 0 to 254
147 b = 255;
148 } else { // Magenta to Red-1
149 r = 255;
150 b = 1530 - hue; // b = 255 to 1
151 }
152 } else { // Last 0.5 Red (quicker than % operator)
153 r = 255;
154 g = b = 0;
155 }
156
157 // Apply saturation and value to R,G,B, pack into 32-bit result:
158 uint32_t v1 = 1 + val; // 1 to 256; allows >>8 instead of /255
159 uint16_t s1 = 1 + sat; // 1 to 256; same reason
160 uint8_t s2 = 255 - sat; // 255 to 0
161 return ((((((r * s1) >> 8) + s2) * v1) & 0xff00) << 8) |
162 (((((g * s1) >> 8) + s2) * v1) & 0xff00) |
163 (((((b * s1) >> 8) + s2) * v1) >> 8);
164}
