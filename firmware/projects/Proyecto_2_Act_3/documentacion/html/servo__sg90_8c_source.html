23 static float h_time;
24 static float duty_cicle;
25 static int16_t deg;
26 deg = 2 * angle + MAX_ANG; // NOTE: adjusted (angle x 2) for the available servos
27 h_time = (float)(deg/ANG_RANGE) + PULSEW_MS;
28 duty_cicle = (float)(h_time/PERIOD_MS) * PERCENT;
