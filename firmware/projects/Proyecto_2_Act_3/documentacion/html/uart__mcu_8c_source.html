107void UartInit(serial_config_t *port_config){
108 uart_config_t uart_config = {
109 .baud_rate = port_config->baud_rate,
110 .data_bits = UART_DATA_8_BITS,
111 .parity = UART_PARITY_DISABLE,
112 .stop_bits = UART_STOP_BITS_1,
113 .flow_ctrl = UART_HW_FLOWCTRL_DISABLE,
114 .source_clk = UART_SCLK_DEFAULT,
115 };
116 switch(port_config->port){
117 case UART_PC:
118 uart_param_config(UART_NUM_0, &uart_config);
119 uart_set_pin(UART_NUM_0, UART_PIN_NO_CHANGE, UART_PIN_NO_CHANGE, UART_PIN_NO_CHANGE, UART_PIN_NO_CHANGE);
120 if(port_config->func_p != UART_NO_INT){
121 uart_pc_isr_p = port_config->func_p;
122 uart_pc_queue = port_config->param_p;
123 xTaskCreate(uart_pc_event_task, "uart_pc_event_task", 2048, NULL, 12, 0);
124 }else{
125 uart_driver_install(UART_NUM_0, RX_BUFFER_SIZE, TX_BUFFER_SIZE, 0, NULL, 0);
126 }
127 break;
128 case UART_CONNECTOR:
129 uart_param_config(UART_NUM_1, &uart_config);
130 uart_set_pin(UART_NUM_1, UART_CONN_TX, UART_CONN_RX, UART_PIN_NO_CHANGE, UART_PIN_NO_CHANGE);
131 if(port_config->func_p != UART_NO_INT){
132 uart_conn_isr_p = port_config->func_p;
133 uart_conn_queue = port_config->param_p;
134 xTaskCreate(uart_conn_event_task, "uart_conn_event_task", 2048, NULL, 12, NULL);
135 }else{
136 uart_driver_install(UART_NUM_1, RX_BUFFER_SIZE, TX_BUFFER_SIZE, 0, NULL, 0);
137 }
138 break;
139 }
140}
