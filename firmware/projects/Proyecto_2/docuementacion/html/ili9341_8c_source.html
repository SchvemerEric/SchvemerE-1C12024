222void WriteLCD(lcd_cmd_t * data){
223 SpiInit(&spi_conf);
224 /* If command is NULL don't send command */
225 if (data->cmd != NULL){
226 /* Send command */
227 GPIOOff(ili9341_dc);
228 SpiWrite(ili9341_spi, &data->cmd, 1);
229 }
230 /* If there are parameters or data to send */
231 if (data->databytes != NULL){
232 /* Send parameters or data */
233 GPIOOn(ili9341_dc);
234 SpiWrite(ili9341_spi, data->data, data->databytes);
235 }
236}

260void Fill(uint16_t x0, uint16_t y0, uint16_t x1, uint16_t y1, uint16_t color){
261 static uint16_t i;
262 static int32_t bytes_count;
263 static int16_t x_dist, y_dist;
264 static uint8_t pixel[MAX_VALUE_SIZE];
265
266 x_dist = x1 - x0;
267 y_dist = y1 - y0;
268 if (x0 > x1){
269 x_dist = - x_dist;
270 }
271 if (y0 > y1){
272 y_dist = - y_dist;
273 }
274 /* Number of bytes to write. We have to write 2 bytes/pixel (16bits color) */
275 bytes_count = (x_dist + 1) * (y_dist + 1) * 2;
276 /* Define area to fill */
277 SetCursorPosition(x0, y0, x1, y1);
278
279 for (i = 0; i < MAX_VALUE_SIZE; i += 2){
280 pixel[i] = HighByte(color);
281 pixel[i + 1] = LowByte(color);
282 }
283 /* Start writing LCD memory */
284 lcd_cmd_t lcd_write = {MEM_WRITE, NULL, NULL};
285 WriteLCD(&lcd_write);
286
287 while(bytes_count - MAX_VALUE_SIZE > 0){
288 lcd_cmd_t lcd_pixel = {NULL, MAX_VALUE_SIZE, pixel};
289 WriteLCD(&lcd_pixel);
290 bytes_count -= MAX_VALUE_SIZE;
291 }
292 lcd_cmd_t lcd_pixel = {NULL, bytes_count, pixel};
293 WriteLCD(&lcd_pixel);
294}

300uint8_t ILI9341Init(spi_dev_t spi_dev, uint8_t gpio_dc, uint8_t gpio_rst){
301 /* SPI configuration */
302 spi_conf.device = spi_dev;
303 ili9341_spi = spi_dev;
304 /* GPIOs configuration and initialization */
305 ili9341_dc = gpio_dc;
306 ili9341_rst = gpio_rst;
307 GPIOInit(ili9341_dc, GPIO_OUTPUT);
308 GPIOInit(ili9341_rst, GPIO_OUTPUT);
309
310 /* RST must be held low for minimum 10µsec after VCC have been applied */
311 DelayUs(10);
312 GPIOOn(ili9341_rst);
313 /* Wait more than 10µsec after RST high before sending a command */
314 DelayUs(10);
315 /* It will be necessary to wait 5msec before sending new command following software reset */
316 WriteLCD(&lcd_reset);
317 DelayMs(5);
318 /* Send initial configuration to LCD */
319 for (uint8_t i = 0; i < sizeof(lcd_init)/sizeof(lcd_cmd_t); i++){
320 WriteLCD(&lcd_init[i]);
321 }
322 /* It will be necessary to wait 5msec before sending next command after sleep out */
323 WriteLCD(&lcd_sleep_out);
324 DelayMs(10);
325 WriteLCD(&lcd_on);
326 DelayMs(20);
327 /* Start screen on White */
328 ILI9341Fill(ILI9341_WHITE);
329 DelayMs(20);
330 return true;
331}

380void ILI9341DrawChar(uint16_t x, uint16_t y, char data, Font_t* font, uint16_t foreground, uint16_t background){
381 static uint16_t i, j, k;
382 static uint16_t char_row;
383 static uint16_t lcd_x, lcd_y;
384 static int32_t bytes_count;
385 static uint8_t pixel[MAX_VALUE_SIZE];
386
387 /* Set coordinates */
388 lcd_x = x;
389 lcd_y = y;
390
391 /* If at the end of a line of display, go to new line and set x to 0 position */
392 if ((lcd_x + font->FontWidth) > lcd_orientation.width) {
393 lcd_y += font->FontHeight;
394 lcd_x = 0;
395 }
396
397 SetCursorPosition(lcd_x, lcd_y, lcd_x + font->FontWidth - 1, lcd_y + font->FontHeight - 1);
398
399 /* Number of bytes to write. We have to write 2 bytes/pixel */
400 bytes_count = font->FontHeight * font->FontWidth * 2;
401
402 /* Start writing LCD memory */
403 lcd_cmd_t lcd_write = {MEM_WRITE, NULL, NULL};
404 WriteLCD(&lcd_write);
405
406 /* Draw font data */
407 /* go through character rows */
408 k = 0;
409 for (i = 0; i < font->FontHeight; i++) {
410 /* each 16bits data of a font character draws a full row of that character */
411 char_row = font->data[(data - ' ') * font->FontHeight + i];
412 /* go through character columns */
413 for (j = 0; j < font->FontWidth; j++) {
414 /* If exceed buffer size, send buffer */
415 if ((2 * j + i * font->FontWidth * 2 - k * MAX_VALUE_SIZE + 1) > MAX_VALUE_SIZE) {
416 lcd_cmd_t lcd_pixels = {NULL, MAX_VALUE_SIZE, pixel};
417 WriteLCD(&lcd_pixels);
418 bytes_count -= MAX_VALUE_SIZE;
419 k++;
420 }
421 /* The n=FontWidth first bits of the 16bits row data draws the corresponding part of a character */
422 if (char_row & (MSK_BIT16 >> j)){
423 /* if bit = 1, draw put foreground color */
424 pixel[2 * j + i * font->FontWidth * 2 - k * MAX_VALUE_SIZE] = HighByte(foreground);
425 pixel[2 * j + i * font->FontWidth * 2 - k * MAX_VALUE_SIZE + 1] = LowByte(foreground);
426 }
427 else{
428 pixel[2 * j + i * font->FontWidth * 2 - k * MAX_VALUE_SIZE] = HighByte(background);
429 pixel[2 * j + i * font->FontWidth * 2 - k * MAX_VALUE_SIZE + 1] = LowByte(background);
430 }
431 }
432 }
433 /* Send the rest of the buffer */
434 lcd_cmd_t lcd_pixels = {NULL, bytes_count, pixel};
435 WriteLCD(&lcd_pixels);
436}

454 static uint16_t lcd_x, lcd_y;
455
456 /* Set coordinates */
457 lcd_x = x;
458 lcd_y = y;
459
460 while (*str != '\0'){ /* End of string */
461 /* New line */
462 if (*str == '\n'){
463 lcd_y += font->FontHeight + 1;
464 /* if after \n is also \r, than go to the left of the screen */
465 if (*(str + 1) == '\r'){
466 lcd_x = 0;
467 str++;
468 }
469 else{
470 lcd_x = x;
471 }
472 str++;
473 }
474 else if (*str == '\r'){
475 str++;
476 }
477
478 /* Put character to LCD */
479 ILI9341DrawChar(lcd_x, lcd_y, *str, font, foreground, background);
480 /* Next character */
481 str++;
482 lcd_x += font->FontWidth;
483 }
484}

499void ILI9341DrawLine(uint16_t x0, uint16_t y0, uint16_t x1, uint16_t y1, uint16_t color){
500 static int16_t x_dist, y_dist, x_grow, y_grow, error, error_2;
501
502 /* Check for overflow */
503 if (x0 >= lcd_orientation.width){
504 x0 = lcd_orientation.width - 1;
505 }
506 if (x1 >= lcd_orientation.width){
507 x1 = lcd_orientation.width - 1;
508 }
509 if (y0 >= lcd_orientation.height){
510 y0 = lcd_orientation.height - 1;
511 }
512 if (y1 >= lcd_orientation.height){
513 y1 = lcd_orientation.height - 1;
514 }
515
516 /* Calculate x y distances and determine grow direction */
517 x_dist = x1 - x0;
518 y_dist = y1 - y0;
519 if (x0 > x1){
520 x_dist = -x_dist;
521 x_grow = LEFT;
522 }
523 else{
524 x_grow = RIGHT;
525 }
526 if (y0 > y1){
527 y_dist = -y_dist;
528 y_grow = UP;
529 }
530 else{
531 y_grow = DOWN;
532 }
533
534 /* Vertical or horizontal line */
535 if (x_dist == 0 || y_dist == 0){
536 Fill(x0, y0, x1, y1, color);
537 }
538 /* Diagonal line */
539 else{
540 if(x_dist > y_dist){
541 error = x_dist / 2;
542 }
543 else{
544 error = y_dist / 2;
545 }
546
547 while (1){
548 /* Draw start point */
549 ILI9341DrawPixel(x0, y0, color);
550 /* Loop ends when start point reaches end point */
551 if (x0 == x1 || y0 == y1){
552 break;
553 }
554 error_2 = error;
555 /* Determine if line must grow in x direction */
556 if (error_2 > -x_dist){
557 error -= y_dist;
558 x0 += x_grow; /* Move start point */
559 }
560 /* Determine if line must grow in y direction */
561 if (error_2 < y_dist){
562 error += x_dist;
563 y0 += y_grow; /* Move start point */
564 }
565 }
566 }
567}

693 }
694 }else if((y2 <= y1) && (y2 <= y0)){

708 }

710 // Bottom flat triangle

719 curx2 += invslope2;
720 scanline_y++;
721 }
722 }
723 else if (y_0 == y_1){
724 // Top flat triangle

732 curx1 -= invslope1;
733 curx2 -= invslope2;
734 scanline_y--;
735 }
736 }
737 else{
738 // Split in a flat top triangle and a bottom flat triangle
739 x_aux = (int)(x_0 + (float)(y_1 - y_0) / (float)(y_2 - y_0) * (x_2 - x_0));
740 y_aux = y_1;
741 // Bottom flat triangle
742 invslope1 = (float)(x_1 - x_0) / (float)(y_1 - y_0);
743 invslope2 = (float)(x_aux - x_0) / (float)(y_aux - y_0);
744 curx1 = x_0;
745 curx2 = x_0;
746 scanline_y = y_0;
747 while(scanline_y < y_1){
748 ILI9341DrawLine((int)curx1, scanline_y, (int)curx2, scanline_y, color);
749 curx1 += invslope1;
750 curx2 += invslope2;
751 scanline_y++;
752 }
753 // Top flat triangle
754 invslope1 = (float)(x_2 - x_1) / (float)(y_2 - y_1);
755 invslope2 = (float)(x_2 - x_aux) / (float)(y_2 - y_aux);
756 curx1 = x_2;
757 curx2 = x_2;
758 scanline_y = y_2;
759 while(scanline_y > y_1){
760 ILI9341DrawLine((int)curx1, scanline_y, (int)curx2, scanline_y, color);
761 curx1 -= invslope1;
762 curx2 -= invslope2;
763 scanline_y--;
764 }
765 ILI9341DrawLine(x_1, y_1, x_aux, y_aux, color);
766 }
