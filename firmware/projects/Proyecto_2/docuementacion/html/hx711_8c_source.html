29uint8_t shiftIn(void)
30{
31 uint8_t value = 0;
32
33 for (uint8_t i = 0; i < 8; ++i)
34 {
35 GPIOOn(internal_pd_sck);//PD_SCK_SET_HIGH;
36 value |= GPIORead(internal_dout) << (7 - i);
37 GPIOOff(internal_pd_sck);//PD_SCK_SET_LOW;
38 }
39 return value;
40}

62void HX711_setGain(uint8_t gain)
63{
64 switch (gain)
65 {
66 case 128: // channel A, gain factor 128
67 GAIN = 1;
68 break;
69 case 64: // channel A, gain factor 64
70 GAIN = 3;
71 break;
72 case 32: // channel B, gain factor 32
73 GAIN = 2;
74 break;
75 }
76
77 GPIOOff(internal_pd_sck);//PD_SCK_SET_LOW;
78 HX711_read();
79}

81uint32_t HX711_read(void)
82{
83 // wait for the chip to become ready
84 while (!HX711_isReady());
85
86 unsigned long count;
87 unsigned char i;
88
89 GPIOOn(internal_dout);//DOUT_SET_HIGH;
90
91 DelayUs(1);
92
93 GPIOOff(internal_pd_sck);//PD_SCK_SET_LOW;
94 DelayUs(1);
95
96 count=0;
97 while(GPIORead(internal_dout));
98 for(i=0;i<24;i++)
99 {
100 GPIOOn(internal_pd_sck);//PD_SCK_SET_HIGH;
101 DelayUs(1);
102 count=count<<1;
103 GPIOOff(internal_pd_sck);//PD_SCK_SET_LOW;
104 DelayUs(1);
105 if(GPIORead(internal_dout))
106 count++;
107 }
108 count = count>>6;
109 GPIOOn(internal_pd_sck);//PD_SCK_SET_HIGH;
110 DelayUs(1);
111 GPIOOff(internal_pd_sck);//PD_SCK_SET_LOW;
112 DelayUs(1);
113 count ^= 0x800000;
114 return(count);
115}
