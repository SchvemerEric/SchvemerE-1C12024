59uint8_t SpiInit(spi_mcu_config_t* spi){
60 static bool spi_initialized = false;
61 if(!spi_initialized){
62 spi_bus_initialize(SPI2_HOST, &bus_cfg, SPI_DMA_CH_AUTO);
63 spi_initialized = true;
64 }
65 spi_device_interface_config_t dev_cfg = {
66 .clock_speed_hz = spi->bitrate,
67 .mode = spi->clk_mode,
68 .queue_size = 8,
69 };
70 switch(spi->device){
71 case SPI_1:
72 dev_cfg.spics_io_num = PIN_NUM_CS1;
73 transfer_mode_1 = spi->transfer_mode;
74 if(transfer_mode_1 == SPI_INTERRUPT){
75 dev_cfg.post_cb = spi_1_isr;
76 }
77 spi_bus_add_device(SPI2_HOST, &dev_cfg, &spi_1);
78 spi_1_isr_p = spi->func_p;
79 spi_1_user_data = spi->param_p;
80 break;
81 case SPI_2:
82 dev_cfg.spics_io_num = PIN_NUM_CS2;
83 if(transfer_mode_2 == SPI_INTERRUPT){
84 dev_cfg.post_cb = spi_2_isr;
85 }
86 transfer_mode_1 = spi->transfer_mode;
87 spi_bus_add_device(SPI2_HOST, &dev_cfg, &spi_2);
88 spi_2_isr_p = spi->func_p;
89 spi_2_user_data = spi->param_p;
90 break;
91 case SPI_3:
92 dev_cfg.spics_io_num = PIN_NUM_CS3;
93 if(transfer_mode_3 == SPI_INTERRUPT){
94 dev_cfg.post_cb = spi_3_isr;
95 }
96 transfer_mode_1 = spi->transfer_mode;
97 spi_bus_add_device(SPI2_HOST, &dev_cfg, &spi_3);
98 spi_3_isr_p = spi->func_p;
99 spi_3_user_data = spi->param_p;
100 break;
101 }
102 return 0;

106 spi_transaction_t t;
107 memset(&t, 0, sizeof(t)); // Zero out the transaction
108 t.length = rx_buffer_size * 8; // tx_buffer_size is in bytes, transaction length is in bits.
109 t.rxlength = rx_buffer_size * 8;
110 t.rx_buffer = rx_buffer; // Data
