67void GPIOInit(gpio_t pin, io_t io){
68 if((pin == GPIO_14) || (pin > GPIO_23)){
69 return;
70 }
71 if(io == GPIO_INPUT){
72 gpio_list[pin].mode = GPIO_MODE_INPUT;
73 } else if(io == GPIO_OUTPUT){
74 gpio_list[pin].mode = GPIO_MODE_OUTPUT;
75 }
76 gpio_reset_pin(gpio_list[pin].pin);
77 gpio_set_direction(gpio_list[pin].pin, gpio_list[pin].mode);
78 gpio_set_pull_mode(gpio_list[pin].pin, gpio_list[pin].pull);
79}
